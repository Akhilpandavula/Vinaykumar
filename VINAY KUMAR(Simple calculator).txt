  <script>
    let expression = "";

    function press(num) {
      expression += num;
      document.getElementById("display").value = expression;
    }

    function calculate() {
      try {
        expression = eval(expression);
        document.getElementById("display").value = expression;
      } catch (error) {
        document.getElementById("display").value = "Error";
        expression = "";
      }
    }

    function clearDisplay() {
      expression = "";
      document.getElementById("display").value = "";
    }
  </script>
